API Usage Examples
curl -X POST "http://localhost:8800/v1/audio/speech" \ -H "Authorization: Bearer YOUR_API_KEY" \ -H "Content-Type: application/json" \ -d '{ "input": "Hello, world!", "voice": "alloy", "response_format": "mp3", "speed": 1.0 }' \ --output speech.mp3
from openai import OpenAI client = OpenAI( base_url="http://localhost:8800/v1", api_key="your-api-key" ) response = client.audio.speech.create( model="tts-1", voice="alloy", input="Hello, world!", response_format="mp3" ) response.stream_to_file("speech.mp3")
const response = await fetch('http://localhost:8800/v1/audio/speech', { method: 'POST', headers: { 'Authorization': 'Bearer YOUR_API_KEY', 'Content-Type': 'application/json' }, body: JSON.stringify({ input: 'Hello, world!', voice: 'alloy', response_format: 'mp3' }) }); const audioBlob = await response.blob(); const audioUrl = URL.createObjectURL(audioBlob); new Audio(audioUrl).play();
